// 一个GdRom由2个密度区组成，低密度区由2个Track组成，高密度区由1-97个Track组成
GdRomDisc {
    Lead_In_0: any
    SD_Program_Area: Map<number, GdRomTrack>
    Lead_Out_0: any
    Lead_In_1: any
    HD_Program_Area: Map<number, GdRomTrack>
    Lead_Out_1: any
}

// 一个Track由2个Index组成，Index00和Index01
GdRomTrack {
    TrackId: number,
    Type:number,
    Length:number,
    Indices:Map<number, GdRomTrackIndex>(2)
}

// 一个Index的数据，可以由1或多个数据片拼合而成
GdRomTrackIndex {
    FAD:number,
    Length:number,
    ContentSources:Array<GdRomTrackIndexPart>
}

// 一个Index的数据片，由一个来源轨道数据(GdiSourceTrackContent)的特定起始位置和特定长度(字节数)来指定
GdRomTrackIndexPart {
    SourceTrackContent: GdiSourceTrackContent,
    SourceTrackOffset: number,
    Length: number
}

GdiSourceImage {
    TrackCount: number,
    Tracks: Array<GdiSourceTrack>

    GdiContentFormat: EnumGdiContentFormat,
    HDAreaPattern: number,
}

EnumGdiContentFormat {
    Fmt_General,
    Fmt_Redump
}

GdiSourceTrack {
    TrackId: number
    LBA: number,
    Type: EnumTrackType,
    SectorSize: number;
    Filename: string;

    Content: GdiSourceTrackContent;
}

EnumTrackType {
    Type_Audio,
    Type_Data
}

GdiSourceTrackContent {
    OwnerTrack: GdiSourceTrack;
    FileLength: number;
    FileSectorLength: number;
    TrackContentFormat: EnumTrackContentFormat;
    Index01Offset: number;
}

EnumTrackContentFormat {
    Fmt_General,
    Fmt_Redump
}
